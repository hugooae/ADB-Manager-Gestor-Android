# Requisitos y dependencias del sistema

# REQUISITOS DEL SISTEMA
# Este script funciona en sistemas Linux/Unix (Bash)

# HERRAMIENTAS REQUERIDAS (deben estar instaladas en el sistema)
bash>=4.0
curl>=7.0         # Para descargar actualizaciones desde GitHub
grep              # Para filtrar información
sed               # Para procesamiento de texto
awk               # Para análisis de datos
cut               # Para manipulación de columnas
sort              # Para ordenar datos
head/tail         # Para lectura de archivos
find              # Para búsqueda de archivos
du                # Para calcular tamaño de directorios
ls                # Para listar archivos
mkdir             # Para crear directorios
cp/rm             # Para operaciones de archivos
date              # Para marcas de tiempo
md5sum            # Para calcular hash MD5
sha256sum         # Para calcular hash SHA256
wc                # Para contar líneas

# REQUISITO PRINCIPAL
adb               # Android Debug Bridge (Android SDK Tools)
                  # Instalar: sudo apt install android-tools-adb

# - Dispositivo Android con depuración USB activada
# - Cable USB para conexión
# - Android 4.1+ recomendado

# sudo apt update
# sudo apt install -y android-tools-adb curl coreutils grep sed awk

# sudo dnf install android-tools curl coreutils grep sed awk
